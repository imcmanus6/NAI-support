1// NAI — AI support that actually knows your product.
2// Drop it in. It reads your docs *and* your live data.
3
4import { defineSupport, briefly, readonly, byDomain } from '@nai/core'
5
6export default defineSupport({
7  knowledge:     briefly.spaces(['help', 'internal']),  // answers from your docs
8  liveData:      readonly(yourDatabase),                // "why can't I access X?" → solved
9  tickets:       briefly.space('support'),              // filed where your team already works
10  recording:     true,                                  // console · network · errors, auto-attached
11  organizations: byDomain(),                            // teammates share a ticket queue
12})
